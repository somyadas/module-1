REPORT ZMAN_PROGRAM5.
TABLES:ZMAN_EMP_TABLE.

DATA:ITAB TYPE TABLE OF ZMAN_EMP_TABLE WITH HEADER LINE.

SELECT * FROM ZMAN_EMP_TABLE INTO TABLE ITAB.
SORT ITAB BY GRADE.
LOOP AT ITAB.
ON CHANGE OF ITAB-GRADE.
ULINE.
WRITE:/ 'Grade:',ITAB-GRADE.
ULINE.
WRITE:/ 'EMPNO',
      20 'ENAME'.

ULINE.
ENDON.
WRITE:/ ITAB-EMPCODE,
      20 ITAB-ENAME.

ENDLOOP.
ULINE.
